1. Se citesc două numere de la tastatură. Se cere afișarea diferenței dintre succesorul primului număr și predecesorul celui de-al doilea.

#include < iostream >
using namespace std;
int main()
{
int a, b;
cin>>a;
cin>>b;
cout<<(a + 1) – ( b – 1);
return 0;
}

3. Se citesc numere naturale nenule până la întâlnirea primului număr egal cu zero. Să se afișeze numărul de numere naturale citit.

#include < iostream >
using namespace std;
int main()
{
int nr, k=0;
cin>>nr;
while ( nr > 0 )
{
k++;
cin>>nr;
}
cout<<"numere citite="<< k;
return 0;
}



4. Se citește un număr natural. Să se afișeze răsturnatul numărului.

#include < iostream >
using namespace std;
int main()
{
int nr, r=0;
cin>>nr;
while ( nr > 0)
{
r=r*10 + nr % 10;
nr= nr / 10;
}
cout<<"rasturnat="<< r;
return 0;
}



5. Să se afișeze valorile naturale cuprinse în intervalul închis de la a la b, unde a și b sunt citite de la tastatură.

#include < iostream >
using namespace std;
int main()
{
int a, b;
cin>>a>>b;
for (int i=a; i<=b; i=i+1)
cout<< i<<" ";
return 0;
}
